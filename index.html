def main(): print("Welcome to the Agrinho System!") print("This program aims to promote awareness about agriculture and environmental importance.") while True: print("\nMain Menu:") print("1. Ecological Footprint Calculator") print("2. Sustainable Agriculture Tips") print("3. Recycling Educational Game") print("4. About Agrinho Project") print("5. Exit") option = input("Choose an option: ") if option == "1": calculate_ecological_footprint() elif option == "2": show_agriculture_tips() elif option == "3": play_recycling_game() elif option == "4": about_project() elif option == "5": print("Thank you for participating in the Agrinho Project! See you!") break else: print("Invalid option. Please choose a number between 1 and 5.") def calculate_ecological_footprint(): print("\nEcological Footprint Calculator") print("Answer the questions below to estimate your environmental impact:") questions = [ "How many meat meals do you consume per week?", "How many kilometers do you drive/ride by car per week?", "How many hours of electricity do you use daily at home?", "How often do you recycle? (1-never, 5-always)" ] answers = [] for question in questions: answer = input(question + " ") answers.append(float(answer)) # Simple ecological footprint calculation footprint = (answers[0] * 0.5) + (answers[1] * 0.2) + (answers[2] * 0.3) + (5 - answers[3]) print(f"\nYour ecological footprint score is: {footprint:.1f}") if footprint < 10: print("Congratulations! You have a sustainable lifestyle!") elif footprint < 20: print("You're on the right track, but can improve in some areas.") else: print("Consider adopting more sustainable practices to reduce your environmental impact.") def show_agriculture_tips(): print("\nSustainable Agriculture Tips:") tips = [ "1. Practice crop rotation to preserve soil health.", "2. Use composting to naturally fertilize your plants.", "3. Consider organic farming without pesticides.", "4. Implement efficient irrigation systems to save water.", "5. Preserve areas of native vegetation on your property." ] for tip in tips: print(tip) print("\nRemember: small actions make a big difference for the planet!") def play_recycling_game(): print("\nRecycling Educational Game") print("Classify each item in the correct bin: Organic, Recyclable, or Waste.") items = { "Banana peel": "Organic", "PET bottle": "Recyclable", "Used battery": "Waste", "Old newspaper": "Recyclable", "Food scraps": "Organic", "Burnt-out light bulb": "Waste" } score = 0 for item, correct_answer in items.items(): answer = input(f"Where to dispose {item}? ").capitalize() if answer == correct_answer: print("Correct!") score += 1 else: print(f"Incorrect. The right answer is {correct_answer}.") print(f"\nYour final score: {score}/{len(items)}") if score == len(items): print("Excellent! You're a recycling expert!") elif score >= len(items)/2: print("Good job! Keep learning about recycling.") else: print("Study more about recycling to help the environment!") def about_project(): print("\nAbout Agrinho Project:") print("Agrinho is an educational program that promotes:") print("- Environmental awareness") print("- Sustainable agricultural practices") print("- Citizenship education") print("- Development of school projects") print("\nThis software is an example of how technology can be used to promote these goals.") if __name__ == "__main__": main()
